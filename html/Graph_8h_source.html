158  friend std::pair<edge_descriptor, bool> edge (vertex_descriptor u, vertex_descriptor v, const Graph& g) {
159  edge_descriptor ed = make_pair(u,v);
160  bool b = false;
161  // if u or v is not in g, b is false
162  if (num_vertices(g) <= u || num_vertices(g) <= v) {
163  b = false;
164  }
165  else {
166  // if u->v is in adjacent vertices to u, b is true
167  pair<adjacency_iterator, adjacency_iterator> adj = adjacent_vertices(u, g);
168  if (adj.second != find(adj.first, adj.second, ed))
169  b = true;
170  }
171  return make_pair(ed, b);}
